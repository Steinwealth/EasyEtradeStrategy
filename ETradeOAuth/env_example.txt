# ETrade OAuth Environment Variables
# Copy this file to .env and fill in your actual credentials

# === SANDBOX CREDENTIALS ===
ETRADE_SANDBOX_KEY=your_sandbox_consumer_key_here
ETRADE_SANDBOX_SECRET=your_sandbox_consumer_secret_here

# === PRODUCTION CREDENTIALS ===
ETRADE_PROD_KEY=your_production_consumer_key_here
ETRADE_PROD_SECRET=your_production_consumer_secret_here

# === USAGE INSTRUCTIONS ===
# 1. Copy this file: cp env_example.txt .env
# 2. Fill in your actual ETrade credentials
# 3. Run: python etrade_oauth_manager.py start sandbox
# 4. For production: python etrade_oauth_manager.py start prod
# 5. Keep alive: python etrade_oauth_manager.py keepalive sandbox --minutes 70
